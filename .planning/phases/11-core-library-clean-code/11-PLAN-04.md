   ```typescript

     public sources: ContextSource[] = [];
     public config: ContextConfig;

     constructor(config: ContextConfig) {
       this.config = config;
     }
   }

   // AFTER
   export class ContextManager {
     private _sources: ContextSource[] = [];

       this.config = config; // Use setter for validation
     }
   }
   ```

3. Add getters that return copies to prevent external mutation:
   ```typescript

     return [...this._sources]; // Return shallow copy
   }

   public get config(): ContextConfig {
     return { ...this._config }; // Return shallow copy
   }
   ```

4. Add setters with validation for mutable fields:
   ```typescript
   public set config(value: ContextConfig) {
     if (!value) {
       throw new Error('ContextConfig cannot be null');
     }
     if (!value.maxTokens || value.maxTokens <= 0) {
       throw new Error('ContextConfig.maxTokens must be positive');
     }
     this._config = value;
   }
   ```

5. Apply @ValidateInput decorator to setters where appropriate:

   ```typescript
       throw new Error('ContextConfig.maxTokens must be positive');
     }
   })
   public set config(value: ContextConfig) {
     this._config = value;
   }
   ```

6. Update all internal and external references to use getters/setters

7. Update tests to use getters/setters instead of direct field access
</action>

- bin/lib/**/*.ts (all class definitions)
</read_first>

<action>
1. Audit all classes for proper access modifiers:
   - Search for methods without access modifiers
   - Search for fields without access modifiers

2. Apply these rules consistently:
   - **private**: Internal state and helper methods not part of public API
   - **protected**: Only when inheritance requires access (document why)
   - **public**: Only for explicit public API surface

3. Add explicit access modifiers to all class members:
   ```typescript
   // BEFORE
   export class Example {
     value: string;  // No modifier
     doSomething() {}  // No modifier
   }

   // AFTER
   export class Example {
     private _value: string;

     public get value(): string { return this._value; }
     public set value(val: string) { this._value = val; }

     public doSomething(): void { ... }

     private helperMethod(): void { ... }  // Explicit private
   }
   ```

4. Convert protected to private where inheritance doesn't require access:
   - Search for `protected` fields/methods
   - If no child class accesses them, make private

5. Update `bin/lib/index.ts` to only export public API:
   - Ensure internal classes/utilities are not exported
   - Document what is considered public API

6. Add TSDoc to all public methods and getters/setters
</action>
